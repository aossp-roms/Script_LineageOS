    $ apt-get install git (On Ubuntu 15.10 and newer is not necesary)
    $ git clone https://github.com/aossp-roms/Script_LineageOS -b master
    $ cd Script_LineageOS
    $ chmod a+x setup.sh LineageSource.sh
    $ ./setup.sh
